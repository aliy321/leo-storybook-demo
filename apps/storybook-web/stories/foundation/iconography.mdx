export const encodeSvg = svg =>
  `data:image/svg+xml;base64,${(typeof window !== 'undefined' ? window.btoa : globalThis.btoa)(svg)}`;

export const downloadSvg = type => encodeSvg(iconSet[type] ?? '');

export const renderIconGallery = items =>
  items.map(item => (
    <IconItem key={item.name} name={item.name}>
      <a href={downloadSvg(item.name)} download={`${item.name}.svg`} target="_blank" rel="noreferrer">
        <span
          aria-label={item.name}
          role="img"
          style={{ display: 'inline-flex', width: 24, height: 24, color: `var(--leo-color-${item.color ?? 'foreground'}, currentColor)` }}
          dangerouslySetInnerHTML={{ __html: getIconSvg(item.name, '24', '24') }}
        />
      </a>
    </IconItem>
  ));
